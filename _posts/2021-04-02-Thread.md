```java
public class MyThread implements Runnable {
	@Override
	public void run() {
		// 쓰레드가 수행해야 할 작업
		try {
			int result = 10 / 0;
			System.out.println(result);
		} catch (ArithmeticException e) {
			System.out.println("예외 발생");
		}
		
	}
}
```
