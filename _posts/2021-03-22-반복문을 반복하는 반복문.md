package chapter4;

public class Ex11 {

	public static void main(String[] args) {
		
		System.out.println("구구단 2단 - 5단");
		System.out.println();
		
		int ii = 2;
		
		while (ii <= 5) {
			int i = 1;
			
			while (i <= 9) {
				System.out.println(ii +" * " + i + " = " + (ii*i));
				i += 1;
        
			} System.out.println(); // end while ii
			ii = ii + 1;
		} // end while i
			
	} // end main

} // end class
