```java
package chapter5;

public enum BirthMonth {
	Jan("1월"), Feb("2월"), Mar("3월"), Apr("4월"), May("5월"),
	Jun("6월"), Jul("7월"), Aug("8월"), Sep("9월"), Oct("10월"),
	Nov("11월"), Dec("12월"); 
	
	private final String korMonth;
	BirthMonth(String korMonth) {
		this.korMonth = korMonth;
	}
	
	public String getkorMonth() {
		return korMonth;
	}
}
```
