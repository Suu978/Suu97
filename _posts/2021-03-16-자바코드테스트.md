```java
public static void main(String[] args) {
		
		byte num1 = 127;
		short num2 = -129;
		int num3 = 32768;
		int num = 704;
		int temp = num;
		temp = 59;
		
		System.out.println(num1);
		System.out.println(num2);
		System.out.println(num3);
		System.out.println(num);
		System.out.println(temp);
		System.out.println();
		
		double fnum = 70.4;
		float fnum1 = 1.2345f;
		double d_value = 3.14;
		d_value = 7.81;
		
		System.out.println(fnum);
		System.out.println(fnum1);
		System.out.println(d_value);
		System.out.println();
		
		// char는 문자뿐만 아니라 1byte크기의 정수형도 저장 가능하며, 문자로 바껴서 출력됨.
		char monja1 = 'u';
		char monja2 = 65;
		System.out.println(monja1);
		System.out.println(monja2);
		System.out.println(); 
		
		String var1 = "Cet";
		String var2 = "메롱";
		System.out.println(var1);
		System.out.println(var2);
		System.out.println();
		
		char var3 = 'A';
		char var4 = '1';
		char var5 = 1;
		String var6 = "오늘은";
		String var7 = "2021년 3월 16일";
		
		System.out.println(var3);
		System.out.println(var4);
		System.out.println(var4 - 1); // 사실은 문자 1의 정수형인 49가 저장되어 있음.
		System.out.println(var5);
		System.out.println(var6);
		System.out.println(var6 + " " +var7);
		System.out.println();
		
		// boolean은 논리값을 저장하는 연산자이다. (참, 거짓 두 종류)
		boolean boo1 = true;
		System.out.println(boo1);
		boo1 = false;
		System.out.println(boo1);
		System.out.println();
		
		// *기본 데이터 타입과 **사용자 정의 데이터 타입 두 가지가 존재함.
		// * 자바 개발자가 만들어 준 데이터의 형태 (String을 뺀 나머지 8개 타입)
		// ** 개발자가 필요에 의해서 만든 데이터 타입 (String)
		// *은 대입 연산자만을 사용해서 데이터를 저장할 수 있다.
		// **은 대입 연산자와 new연산자를 사용해서 데이터를 저장할 수 있다.
		String var8 = new String("new연산자를 사용해서 데이터 저장하기");
		System.out.println(var8);
	}
  ```
